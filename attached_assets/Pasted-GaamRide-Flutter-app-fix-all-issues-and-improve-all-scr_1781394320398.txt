=== ISSUE 1: PROFILE SCREEN STUCK ON LOADING ===

The profile screen loads forever. Fix this:

In customer/customer_profile_screen.dart,
the _load() function calls RideService.getCustomerHistory(uid)
which requires a Firestore composite index.

Fix _load() to handle errors gracefully:

```dart
Future<void> _load() async {
  final uid = FirebaseAuth.instance.currentUser?.uid;
  if (uid == null) return;

  try {
    // Load user and history in parallel but catch separately
    final userDoc = await FirebaseFirestore.instance
        .collection('users').doc(uid).get();

    if (!mounted) return;

    if (userDoc.exists) {
      setState(() {
        _user = UserModel.fromFirestore(userDoc);
      });
    }

    // Load history separately — don't block profile if this fails
    try {
      final rideDocs = await FirebaseFirestore.instance
          .collection('rides')
          .where('customerId', isEqualTo: uid)
          .orderBy('createdAt', descending: true)
          .limit(10)
          .get();

      if (!mounted) return;
      setState(() {
        _rides = rideDocs.docs;
        _totalSpent = _rides.fold(0.0,
          (sum, doc) => sum + ((doc.data()['fare'] ?? 0).toDouble()));
      });
    } catch (historyError) {
      // History failed (maybe index not created yet) — show profile anyway
      debugPrint('History load error: $historyError');
      // Show message to create index
      setState(() { _historyError = true; });
    }

  } catch (e) {
    debugPrint('Profile load error: $e');
    if (!mounted) return;
    setState(() { _loadError = true; });
  } finally {
    if (mounted) setState(() { _isLoading = false; });
  }
}
```

In otp_screen.dart and login_screen.dart _handlePostLogin():

```dart
Future<void> _handlePostLogin(String uid, String loginRole) async {
  final doc = await FirebaseFirestore.instance
      .collection('users').doc(uid).get();

  if (!mounted) return;

  if (!doc.exists) {
    // Brand new user — go to setup based on login choice
    if (loginRole == 'saathi') {
      _goRemoveAll(const SaathiRegisterScreen());

      _goRemoveAll(const SaathiMainShell());
```
