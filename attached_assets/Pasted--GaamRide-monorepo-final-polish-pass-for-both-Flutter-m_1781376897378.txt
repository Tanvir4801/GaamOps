```jsx
// Listen to rides + haul_bookings ordered by createdAt
// Show last 10 events with:
// - Color coded dot (green=ride, orange=haul)

// Auto-scroll to top when new event arrives
useEffect(() => {

    (snap) => setRideEvents(snap.docs)
```

### 3. Revenue Page — fix calculations

```jsx
// Read from CORRECT collections:
// rides (not bookings) for GaamRide revenue
// haul_bookings for GaamHaul commission

const calculateRevenue = async (dateRange) => {
  // GaamRide: sum of fare from completed rides
  const ridesSnap = await getDocs(query(
    collection(db, 'rides'),
    where('status', '==', 'completed'),
    where('createdAt', '>=', dateRange.start),
    where('createdAt', '<=', dateRange.end)
  ));
  const rideRevenue = ridesSnap.docs.reduce(
    (sum, doc) => sum + (doc.data().fare || 0), 0);

  // GaamHaul: 75 × completed bookings
  const haulSnap = await getDocs(query(
    collection(db, 'haul_bookings'),
    where('status', '==', 'completed'),
    where('createdAt', '>=', dateRange.start),
    where('createdAt', '<=', dateRange.end)
  ));
  const haulRevenue = haulSnap.docs.length * 75;

  return { rideRevenue, haulRevenue,
    total: rideRevenue + haulRevenue };
};
```

### 10. Export CSV function

```jsx
// Add to Ride History + Revenue pages
const exportCSV = (data, filename) => {
  const headers = Object.keys(data[0]);
  const rows = data.map(row =>
    headers.map(h => `"${row[h] || ''}"`).join(','));
  const csv = [headers.join(','), ...rows].join('\n');
  const blob = new Blob([csv], { type: 'text/csv' });
  const url = URL.createObjectURL(blob);
  const a = document.createElement('a');
  a.href = url;
  a.download = `${filename}_${new Date().toISOString().split('T')[0]}.csv`;
  a.click();
};
```
