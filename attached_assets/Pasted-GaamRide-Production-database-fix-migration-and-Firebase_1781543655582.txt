// ─── FIX 1: User roles ──────────────────────────
async function fixUserRoles() {
  console.log('📝 Fixing user roles...');
  const snap = await db.collection('users').get();
  const batch = db.batch();
  let fixed = 0;

  for (const doc of snap.docs) {
    const role = doc.data().role;
    
    // Fix pipe-separated roles
    if (role && role.includes('|')) {
      // "customer | saathi | both" → pick the most specific role
      let correctRole = 'customer';
      if (role.includes('saathi') && role.includes('customer')) {
        correctRole = 'both';
      } else if (role.includes('saathi')) {
        correctRole = 'saathi';
      }
      
      batch.update(doc.ref, {
        role: correctRole,
        updatedAt: admin.firestore.FieldValue.serverTimestamp()
      });
      console.log(`  Fixed user ${doc.id}: "${role}" → "${correctRole}"`);
      fixed++;
    }

    // Fix missing fields
    const updates = {};
    if (!doc.data().isBlocked) updates.isBlocked = false;
    if (!doc.data().fcmToken) updates.fcmToken = '';
    if (!doc.data().profilePhoto) updates.profilePhoto = '';
    if (!doc.data().displayName && doc.data().name) {
      updates.displayName = doc.data().name;
    }
    
    if (Object.keys(updates).length > 0) {
      batch.update(doc.ref, updates);
    }
  }

  await batch.commit();
  console.log(`  ✅ Fixed ${fixed} user role(s)\n`);
}

// ─── FIX 3: Haul bookings ───────────────────────
async function fixHaulBookings() {
  console.log('🚛 Fixing haul_bookings...');
  const snap = await db.collection('haul_bookings').get();
  let fixed = 0;

  for (const doc of snap.docs) {
    const data = doc.data();
    const updates = {};
    let needsFix = false;

    // Fix pipe status
    if (data.status && data.status.includes('|')) {
      updates.status = 'cancelled';
      needsFix = true;
    }

    // Fix pipe duration
    if (data.duration && data.duration.includes('|')) {
      updates.duration = '1h';
      updates.durationHours = 1;
      needsFix = true;
    }

    // Fix pipe vehicleType
    if (data.vehicleType && data.vehicleType.includes('|')) {
      updates.vehicleType = 'mini_tempo';
      needsFix = true;
    }

    // Fix double-colon field names
    // Firestore doesn't allow renaming fields directly
    // We need to read, delete old field, set new field
    const rawData = doc.data();
    
    // Check for "customerId::" field (double colon bug)
    if ('customerId::' in rawData) {
      updates['customerId'] = rawData['customerId::'] || '';
      // We can't delete fields with illegal names via batch
      // so we use set with merge to overwrite the whole doc
      needsFix = true;
      console.log(`  Found double-colon field in ${doc.id}`);
    }

    // Fix missing bookingId
    if (!data.bookingId || data.bookingId === '') {
      updates.bookingId = doc.id;
      needsFix = true;
    }

    if (needsFix) {
      // Use set with merge to handle the double-colon field
      await doc.ref.set({
        ...rawData,
        ...updates,
        // Remove the bad field by not including it
        'customerId::': admin.firestore.FieldValue.delete(),
        'customerName::': admin.firestore.FieldValue.delete(),
      }, { merge: true });
      
      fixed++;
      console.log(`  Fixed haul_booking ${doc.id}`);
    }
  }

  console.log(`  ✅ Fixed ${fixed} haul_booking(s)\n`);
}

async function fixSaathisPosition() {
  console.log('📍 Fixing saathi positions...');
  const snap = await db.collection('saathis').get();
  const batch = db.batch();
  let fixed = 0;

  for (const doc of snap.docs) {
    const data = doc.data();
    const pos = data.position;
    
    // Check if position has San Francisco coordinates
    // SF is around 37.4N, 122.0W
    if (pos && pos.geopoint) {
      const lat = pos.geopoint._latitude || pos.geopoint.latitude || 0;
      const lng = pos.geopoint._longitude || pos.geopoint.longitude || 0;
      
      if (Math.abs(lat - 37.42) < 0.1 && Math.abs(lng - (-122.08)) < 0.1) {

          position: {
            geohash: '',
            geopoint: new admin.firestore.GeoPoint(0, 0)
          },
          isAvailable: false,
          isOnline: false,
        });
        console.log(`  Reset SF coordinates for saathi ${doc.id}`);
        fixed++;
      }
    }

    // Fix fcmToken null → empty string
    if (data.fcmToken === null || data.fcmToken === undefined) {
      batch.update(doc.ref, { fcmToken: '' });
    }
  }

  await batch.commit();
  console.log(`  ✅ Fixed ${fixed} saathi position(s)\n`);
}

      'searching', 'accepted', 'started', 'completed', 'cancelled'
    ];
    
    final duration = data['duration'] as String? ?? '';
    final vehicle = data['vehicleType'] as String? ?? '';
    final status = data['status'] as String? ?? '';
    
    assert(!duration.contains('|'),
      'INVALID duration contains pipe: $duration');
    assert(validDurations.contains(duration),
      'INVALID duration: $duration');
    assert(!vehicle.contains('|'),
      'INVALID vehicleType contains pipe: $vehicle');
    assert(validVehicles.contains(vehicle),
      'INVALID vehicleType: $vehicle');
    assert(!status.contains('|'),
      'INVALID status contains pipe: $status');
    assert(validStatuses.contains(status),
      'INVALID status: $status');
    assert(data['appCommission'] == 75,
      'INVALID: appCommission must be 75');
  }
  
  static void validateUserData(Map<String, dynamic> data) {
    const validRoles = ['customer', 'saathi', 'both'];
    final role = data['role'] as String? ?? '';
    
    assert(!role.contains('|'),
      'INVALID role contains pipe: $role');
    assert(validRoles.contains(role),
      'INVALID role: $role');
    assert(data['uid'] is String,
      'INVALID: missing uid');
  }
}

class AuthService {

    if (!context.mounted) return;
    
    if (!userDoc.exists) {
      // Check if coming from saathi login flow

      final loginRole = prefs.getString('login_role') ?? 'customer';
      
      if (loginRole == 'saathi') {
        Navigator.pushAndRemoveUntil(context,
          MaterialPageRoute(
            builder: (_) => const SaathiRegistrationScreen()),
          (route) => false);
      } else {
        Navigator.pushAndRemoveUntil(context,
          MaterialPageRoute(
            builder: (_) => const CustomerRegistrationScreen()),
          (route) => false);
      }
      return;
    }
    
    // User exists
    final data = userDoc.data()!;
    
    // Check blocked
    if (data['isBlocked'] == true) {
      await FirebaseAuth.instance.signOut();
      if (!context.mounted) return;
      showDialog(context: context, builder: (_) => AlertDialog(
        title: Text('Account Blocked'),
        content: Text('તમારું account block થયેલ છે. '
          'Admin નો સંપર્ક કરો.'),
        actions: [
          TextButton(
            onPressed: () => Navigator.pop(context),
            child: Text('OK'))],
      ));
      return;
    }
    
    // Get role — fix if corrupted
    String role = data['role'] as String? ?? 'customer';
    if (role.contains('|')) {
      // Fix corrupted role
      if (role.contains('saathi') && role.contains('customer')) {
        role = 'both';
      } else if (role.contains('saathi')) {
        role = 'saathi';
      } else {
        role = 'customer';
      }
      // Update in Firestore
      await FirebaseFirestore.instance
          .collection('users').doc(uid)
          .update({'role': role});
    }
    
    if (!context.mounted) return;
    
    if (role == 'saathi') {
      Navigator.pushAndRemoveUntil(context,
        MaterialPageRoute(builder: (_) => const SaathiMainShell()),
        (route) => false);
    } else {
      Navigator.pushAndRemoveUntil(context,
        MaterialPageRoute(
          builder: (_) => const CustomerMainShell()),
        (route) => false);
    }
  }
  
  static Future<void> _migrateTempSaathiIfExists(
    String uid, String phone) async {
    
    try {
      final phoneClean = phone.replaceAll('+91', '').trim();
      
      // Check if admin created a temp doc with phone as ID
      final tempDoc = await FirebaseFirestore.instance
          .collection('saathis')
          .doc(phoneClean)
          .get();
      
      if (!tempDoc.exists) return;
      
      final tempData = tempDoc.data()!;
      // Only migrate if uid looks like a phone number (temp doc)
      if (tempData['uid'] != phoneClean) return;
      
      print('📦 Migrating admin-created saathi doc...');
      
      final batch = FirebaseFirestore.instance.batch();
      
      // Create proper doc with real uid
      batch.set(
        FirebaseFirestore.instance.collection('saathis').doc(uid),
        {
          ...tempData,
          'uid': uid,
          'phone': phone,
          'updatedAt': FieldValue.serverTimestamp(),
        });
      
      // Create users doc
      batch.set(
        FirebaseFirestore.instance.collection('users').doc(uid),
        {
          'uid': uid,
          'name': tempData['name'] ?? '',
          'displayName': tempData['name'] ?? '',
          'phone': phone,
          'role': 'saathi',
          'village': tempData['village'] ?? '',
          'profilePhoto': '',
          'fcmToken': '',
          'isBlocked': false,
          'createdAt': FieldValue.serverTimestamp(),
          'updatedAt': FieldValue.serverTimestamp(),
        });
      
      // Delete temp doc
      batch.delete(tempDoc.ref);
      
      await batch.commit();
      print('✅ Migration complete: $phoneClean → $uid');
      
    } catch (e) {
      print('⚠️ Migration error (non-fatal): $e');
    }
  }
}
