CRITICAL BUG 2 — FIREBASE CONNECTION CHECK

Check every screen connects to correct Firestore collection.
Add this diagnostic service that runs on app start:

```dart
// lib/services/firebase_check_service.dart

class FirebaseCheckService {

  static Future<void> runChecks() async {
    if (!kDebugMode) return;

    debugPrint('\n🔍 GaamRide Firebase Connection Check\n');

    await _checkCollection('users');
    await _checkCollection('saathis');
    await _checkCollection('rides');
    await _checkCollection('haul_bookings');
    await _checkCollection('villages');
    await _checkCollection('app_settings');
    await _checkAppSettings();
    await _checkVillages();

    debugPrint('\n✅ Firebase Check Complete\n');
  }

  static Future<void> _checkCollection(String name) async {
    try {
      final snap = await FirebaseFirestore.instance
          .collection(name)
          .limit(1)
          .get();
      debugPrint(
          '✅ $name: connected (${snap.docs.length} docs visible)');
    } catch (e) {
      debugPrint('❌ $name: ERROR — $e');
    }
  }

  static Future<void> _checkAppSettings() async {
    try {
      final doc = await FirebaseFirestore.instance
          .collection('app_settings')
          .doc('config')
          .get();
      if (!doc.exists) {
        debugPrint('⚠️  app_settings/config: document NOT FOUND');
        return;
      }
      final d = doc.data()!;
      debugPrint('✅ app_settings: '
          'fare=₹${d['rideFareBase']}+₹${d['rideFarePerKm']}/km, '
          'haul=₹${d['haulCommission']}, '
          'maintenance=${d['maintenanceMode']}');
    } catch (e) {
      debugPrint('❌ app_settings: $e');
    }
  }

  static Future<void> _checkVillages() async {
    try {
      final snap = await FirebaseFirestore.instance
          .collection('villages')
          .where('isActive', isEqualTo: true)
          .get();
      final names = snap.docs.map((d) => d.data()['name']).join(', ');
      debugPrint('✅ villages: ${snap.docs.length} active — $names');
    } catch (e) {
      debugPrint('❌ villages: $e');
    }
  }
}
```

```dart
// lib/services/settings_service.dart

class AppSettings {
  final double rideFareBase;
  final double rideFarePerKm;
  final double rideFareMinimum;
  final double rideFareMaximum;
  final double haulCommission;
  final bool maintenanceMode;
  final double serviceZoneSWLat;
  final double serviceZoneSWLng;
  final double serviceZoneNELat;
  final double serviceZoneNELng;

  const AppSettings({
    this.rideFareBase = 20,
    this.rideFarePerKm = 8,
    this.rideFareMinimum = 25,
    this.rideFareMaximum = 200,
    this.haulCommission = 75,
    this.maintenanceMode = false,
    this.serviceZoneSWLat = 20.780,
    this.serviceZoneSWLng = 73.190,
    this.serviceZoneNELat = 20.920,
    this.serviceZoneNELng = 73.320,
  });

  factory AppSettings.fromFirestore(Map<String, dynamic> d) {
    final sw = d['serviceZoneSW'] as Map<String, dynamic>? ?? {};
    final ne = d['serviceZoneNE'] as Map<String, dynamic>? ?? {};
    return AppSettings(
      rideFareBase: (d['rideFareBase'] ?? 20).toDouble(),
      rideFarePerKm: (d['rideFarePerKm'] ?? 8).toDouble(),
      rideFareMinimum: (d['rideFareMinimum'] ?? 25).toDouble(),
      rideFareMaximum: (d['rideFareMaximum'] ?? 200).toDouble(),
      haulCommission: (d['haulCommission'] ?? 75).toDouble(),
      maintenanceMode: d['maintenanceMode'] ?? false,
      serviceZoneSWLat: (sw['lat'] ?? 20.780).toDouble(),
      serviceZoneSWLng: (sw['lng'] ?? 73.190).toDouble(),
      serviceZoneNELat: (ne['lat'] ?? 20.920).toDouble(),
      serviceZoneNELng: (ne['lng'] ?? 73.320).toDouble(),
    );
  }
}

class SettingsService {
  static AppSettings? _cached;

  static Future<AppSettings> getSettings() async {
    if (_cached != null) return _cached!;
    try {
      final doc = await FirebaseFirestore.instance
          .collection('app_settings')
          .doc('config')
          .get();
      if (doc.exists && doc.data() != null) {
        _cached = AppSettings.fromFirestore(doc.data()!);
        return _cached!;
      }
    } catch (e) {
      debugPrint('⚠️ Settings load failed, using defaults: $e');
    }
    _cached = const AppSettings();
    return _cached!;
  }

  static void clearCache() => _cached = null;

  // Listen for maintenance mode changes
  static Stream<bool> maintenanceModeStream() {
    return FirebaseFirestore.instance
        .collection('app_settings')
        .doc('config')
        .snapshots()
        .map((snap) => snap.data()?['maintenanceMode'] ?? false);
  }
}
```

```dart
// In customer_home_screen.dart _findSaathi():
// This is the most important query — verify it works

Future<void> _findSaathi() async {
  if (_selectedVillage == null) return;
  if (_userLat == 0 || _userLng == 0) {
    _showSnack('સ્થાન મળ્યું નથી. Refresh કરો.');
    return;
  }

  setState(() { _isSearching = true; _searched = false; });

  try {
    // Step 1: Get all available saathis within 10km
    // Using geoflutterfire_plus for geo query
    final geo = GeoFlutterFire();
    final collectionRef = FirebaseFirestore.instance
        .collection('saathis');

    final center = geo.point(
      latitude: _userLat, longitude: _userLng);

    final stream = geo.collection(collectionRef: collectionRef)
        .within(
          center: center,
          radius: 10, // 10km radius
          field: 'position',
          strictMode: false);

    final docs = await stream.first;

    // Step 2: Filter in Dart (avoids composite index requirement)
    final available = docs
        .where((doc) {
          final d = doc.data() as Map<String, dynamic>;
          return d['isAvailable'] == true &&
                 d['isBlocked'] != true &&
                 d['isOnline'] == true;
        })
        .map((doc) {
          final d = doc.data() as Map<String, dynamic>;
          return SaathiModel.fromMap(d);
        })
        .toList();

    // Step 3: Smart sort
    // a. Same village first (PREFERRED)
    // b. Then by distance
    available.sort((a, b) {
      final aIsSameVillage = a.village == _currentVillageName;
      final bIsSameVillage = b.village == _currentVillageName;

      if (aIsSameVillage && !bIsSameVillage) return -1;
      if (!aIsSameVillage && bIsSameVillage) return 1;

      // Both same or both different — sort by distance
      final aDist = Geolocator.distanceBetween(
        _userLat, _userLng,
        a.lat ?? 0, a.lng ?? 0);
      final bDist = Geolocator.distanceBetween(
        _userLat, _userLng,
        b.lat ?? 0, b.lng ?? 0);
      return aDist.compareTo(bDist);
    });

    if (!mounted) return;
    setState(() {
      _saathiList = available;
      _isSearching = false;
      _searched = true;
    });

    if (available.isEmpty) {
      _showSnack('આ વિસ્તારમાં કોઈ સાથી ઉપલબ્ધ નથી');
    }

  } catch (e) {
    debugPrint('❌ Saathi search error: $e');
    if (!mounted) return;
    setState(() { _isSearching = false; _searched = true; });
    _showSnack('Error: $e');
  }
}
```

```dart
// In saathi_dashboard.dart _toggleOnline():
Future<void> _toggleOnline() async {
  final uid = FirebaseAuth.instance.currentUser?.uid;
  if (uid == null) return;

  if (!_isOnline) {
    // Going ONLINE
    final permission = await Geolocator.requestPermission();
    if (permission == LocationPermission.denied ||
        permission == LocationPermission.deniedForever) {
      _showSnack('Location permission needed to go online');
      return;
    }

    setState(() => _isOnline = true);

    // Get current position
    try {
      final pos = await Geolocator.getCurrentPosition(
        locationSettings: const LocationSettings(
          accuracy: LocationAccuracy.high));

      final geoPoint = GeoFirePoint(pos.latitude, pos.longitude);

      // Update Firestore — verify this works
      await FirebaseFirestore.instance
          .collection('saathis')
          .doc(uid)
          .update({
        'isAvailable': true,
        'isOnline': true,
        'position': geoPoint.data,
        'lastSeen': FieldValue.serverTimestamp(),
      });

      debugPrint('✅ Saathi online, position updated: '
          '${pos.latitude}, ${pos.longitude}');

      // Start location stream
      _startLocationStream(uid);

    } catch (e) {
      debugPrint('❌ Go online error: $e');
      setState(() => _isOnline = false);
      _showSnack('Error: $e');
    }

  } else {
    // Going OFFLINE
    _locationStream?.cancel();

    await FirebaseFirestore.instance
        .collection('saathis')
        .doc(uid)
        .update({
      'isAvailable': false,
      'isOnline': false,
      'lastSeen': FieldValue.serverTimestamp(),
    });

    setState(() => _isOnline = false);
    debugPrint('✅ Saathi offline');
  }
}

void _startLocationStream(String uid) {
  _locationStream = Geolocator.getPositionStream(
    locationSettings: const LocationSettings(
      accuracy: LocationAccuracy.high,
      distanceFilter: 20,
    ),
  ).listen((pos) async {
    if (!_isOnline) return;

    final geoPoint = GeoFirePoint(pos.latitude, pos.longitude);

    try {
      await FirebaseFirestore.instance
          .collection('saathis')
          .doc(uid)
          .update({
        'position': geoPoint.data,
        'lastSeen': FieldValue.serverTimestamp(),
      });
    } catch (e) {
      debugPrint('⚠️ Location update error: $e');
    }
  });
}
```
