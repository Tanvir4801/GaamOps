  @override
  void initState() {
    super.initState();

    // Logo animation
    _logoCtrl = AnimationController(

      duration: const Duration(milliseconds: 900));

    _logoScale = Tween<double>(begin: 0.3, end: 1.0).animate(
      CurvedAnimation(parent: _logoCtrl, curve: Curves.elasticOut));

    _logoFade = Tween<double>(begin: 0.0, end: 1.0).animate(
      CurvedAnimation(parent: _logoCtrl,
        curve: const Interval(0.0, 0.5, curve: Curves.easeIn)));

    // Tagline animation
    _taglineCtrl = AnimationController(
      vsync: this,
      duration: const Duration(milliseconds: 500));

    _taglineFade = Tween<double>(begin: 0.0, end: 1.0).animate(
      CurvedAnimation(parent: _taglineCtrl, curve: Curves.easeIn));

    _taglineSlide = Tween<Offset>(
      begin: const Offset(0, 0.3),
      end: Offset.zero,
    ).animate(CurvedAnimation(
      parent: _taglineCtrl, curve: Curves.easeOut));

    _startSequence();
  }

  Future<void> _startSequence() async {
    // Step 1: Logo appears
    await _logoCtrl.forward();

    // Step 2: Show rotating taglines
    await Future.delayed(const Duration(milliseconds: 300));

    for (int i = 0; i < _taglines.length; i++) {
      if (!mounted) return;
      setState(() {
        _currentTagline = i;
        _showTagline = true;
      });
      _taglineCtrl.forward(from: 0);
      await Future.delayed(const Duration(milliseconds: 1200));
      await _taglineCtrl.reverse();
    }

    // Step 3: Navigate
    await Future.delayed(const Duration(milliseconds: 400));
    if (!mounted) return;
    _navigate();
  }

  Future<void> _navigate() async {
    // Check maintenance mode
    final settings = await SettingsService.getSettings();
    if (!mounted) return;

    if (settings.maintenanceMode) {
      _go(const MaintenanceScreen());
      return;
    }

    final user = FirebaseAuth.instance.currentUser;
    if (user == null) {
      _go(const WelcomeScreen());
      return;
    }

    try {
      final doc = await FirebaseFirestore.instance
          .collection('users').doc(user.uid).get();

      if (!mounted) return;

      if (!doc.exists) {
        _go(const WelcomeScreen());
        return;
      }

      final role = doc.data()!['role'] as String? ?? 'customer';
      if (role.contains('|')) {
        _go(const WelcomeScreen());
        return;
      }

      if (role == 'saathi') {
        _go(const SaathiMainShell());
      } else if (role == 'haul_owner') {
        _go(const VehicleOwnerDashboard());
      } else {
        _go(const CustomerMainShell());
      }
    } catch (_) {
      _go(const WelcomeScreen());
    }
  }

  void _go(Widget screen) {
    Navigator.pushReplacement(context,
      MaterialPageRoute(builder: (_) => screen));
  }

// MarqueeText widget:
class _MarqueeText extends StatefulWidget {
  final List<String> texts;
  const _MarqueeText({required this.texts});
  @override
  State<_MarqueeText> createState() => _MarqueeTextState();
}

class _MarqueeTextState extends State<_MarqueeText> {
  int _current = 0;

  @override
  void initState() {
    super.initState();
    Future.delayed(Duration.zero, _cycle);
  }

  void _cycle() async {
    while (mounted) {
      await Future.delayed(Duration(seconds: 3));
      if (!mounted) return;
      setState(() {
        _current = (_current + 1) % widget.texts.length;
      });
    }
  }

  @override
  Widget build(BuildContext context) {
    return AnimatedSwitcher(
      duration: Duration(milliseconds: 400),
      transitionBuilder: (child, anim) => FadeTransition(
        opacity: anim, child: child),
      child: Text(
        widget.texts[_current],
        key: ValueKey(_current),
        style: TextStyle(
          fontSize: 12,
          color: Color(0xFF1B5E20),
          fontWeight: FontWeight.w500),
        overflow: TextOverflow.ellipsis),
    );
  }
}
