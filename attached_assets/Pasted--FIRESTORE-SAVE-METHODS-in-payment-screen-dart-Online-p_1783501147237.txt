
═══ FIRESTORE SAVE METHODS (in payment_screen.dart) ═══

// Online payment confirmed by Razorpay
Future _saveOnlinePayment(String paymentId) async {
  try {
    await FirebaseFirestore.instance
        .collection('rides')
        .doc(widget.rideId)
        .update({
      'paymentMethod': _selectedMethod,   // 'gpay','phonepe','paytm','upi'
      'paymentStatus': 'paid',
      'paymentId': paymentId,             // Razorpay payment ID
      'paidAt': FieldValue.serverTimestamp(),
    });

    setState(() => _isProcessing = false);
    _showSuccessAndNavigate();

  } catch (e) {
    setState(() => _isProcessing = false);
    ScaffoldMessenger.of(context).showSnackBar(
      SnackBar(content: Text('Saved payment failed: $e')),
    );
  }
}

// Cash payment — no gateway
Future _saveCashPayment() async {
  await FirebaseFirestore.instance
      .collection('rides')
      .doc(widget.rideId)
      .update({
    'paymentMethod': 'cash',
    'paymentStatus': 'paid',   // cash is always "paid" (honour system)
    'paymentId': '',
    'paidAt': FieldValue.serverTimestamp(),
  });

  setState(() => _isProcessing = false);
  _showSuccessAndNavigate();
}

// Navigate to success screen
void _showSuccessAndNavigate() {
  showDialog(
    context: context,
    barrierDismissible: false,
    builder: (_) => AlertDialog(
      shape: RoundedRectangleBorder(borderRadius: BorderRadius.circular(16)),
      content: Column(
        mainAxisSize: MainAxisSize.min,
        children: [
          const Icon(Icons.check_circle, color: Colors.green, size: 60),
          const SizedBox(height: 12),
          const Text('Payment done!', style: TextStyle(fontSize: 18, fontWeight: FontWeight.w600)),
          const SizedBox(height: 6),
          Text('₹${widget.fare.toStringAsFixed(0)} paid', style: const TextStyle(color: Colors.grey)),
          const SizedBox(height: 20),
          SizedBox(
            width: double.infinity,
            child: ElevatedButton(
              style: ElevatedButton.styleFrom(backgroundColor: Color(0xFFf97316), foregroundColor: Colors.white),
              onPressed: () {
                Navigator.of(context).popUntil((r) => r.isFirst);
              },
              child: const Text('Back to home'),
            ),
          ),
        ],
      ),
    ),
  );
}
