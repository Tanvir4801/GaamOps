I am building GaamRide Flutter app. My Firestore backend is messy 
with duplicate collections, wrong field names, and inconsistent 
structure. I need a complete automated migration and cleanup.

## CURRENT PROBLEMS I CAN SEE IN FIRESTORE

1. TWO saathi collections: `saathi` (phone as docId) 
   AND `saathis` (uid as docId) — duplicate, causes confusion
2. `locations` collection — unnecessary, duplicates position data
3. `driver_requests` collection — should not exist separately,
   this data belongs inside bookings
4. `villages` collection has wrong coordinates (lat: 22.92 
   instead of correct 20.83)
5. `bookings` uses a `type` field ('ride' or 'haul') to 
   mix two different things — confusing
6. `users` collection missing important fields
7. `saathi` collection uses phone number as document ID 
   instead of Firebase Auth UID

## STEP 3 — Merge old saathi collection into saathis

```dart
static Future<void> _mergeOldSaathiCollection() async {
  
  // Read all docs from old `saathi` collection (phone as ID)
  final oldDocs = await FirebaseFirestore.instance
      .collection('saathi')
      .get();
  
  if (oldDocs.docs.isEmpty) {
    debugPrint('No old saathi docs to migrate');
    return;
  }
  
  final batch = FirebaseFirestore.instance.batch();
  
  for (final doc in oldDocs.docs) {
    final data = doc.data();
    
    // Try to find matching user by phone
    final phone = data['phone'] as String? ?? doc.id;
    final usersQuery = await FirebaseFirestore.instance
        .collection('users')
        .where('phone', isEqualTo: phone)
        .limit(1)
        .get();
    
    if (usersQuery.docs.isNotEmpty) {
      final uid = usersQuery.docs.first.id;
      
      // Write to saathis collection with uid as doc ID
      final newRef = FirebaseFirestore.instance
          .collection('saathis')
          .doc(uid);
      
      batch.set(newRef, {
        ...data,
        'uid': uid,
        'isAvailable': data['isAvailable'] ?? false,
        'isOnline': data['isOnline'] ?? false,
        'rating': (data['rating'] ?? 5).toDouble(),
        'totalRides': data['totalRides'] ?? 0,
        'updatedAt': FieldValue.serverTimestamp(),
      }, SetOptions(merge: true));
    }
  }
  
  await batch.commit();
  debugPrint('✅ Old saathi collection merged into saathis');
  
  // NOTE: Do NOT delete old saathi collection here
  // Delete manually from Firebase Console after verifying
  // old collection: saathi
  // delete collection: locations
  // delete collection: driver_requests  
}
```

## STEP 5 — Migrate old bookings to rides collection

```dart
static Future<void> _migrateBooingCollection() async {
  
  // Read old bookings collection
  final oldBookings = await FirebaseFirestore.instance
      .collection('bookings')
      .get();
  
  if (oldBookings.docs.isEmpty) return;
  
  final batch = FirebaseFirestore.instance.batch();
  
  for (final doc in oldBookings.docs) {
    final data = doc.data();
    final type = data['type'] as String? ?? 'ride';
    
    if (type == 'ride') {
      // Move to rides collection
      final rideRef = FirebaseFirestore.instance
          .collection('rides')
          .doc(doc.id);
      
      batch.set(rideRef, {
        'rideId': doc.id,
        'customerId': data['userId'] ?? data['customerId'] ?? '',
        'customerName': data['customerName'] ?? '',
        'customerPhone': data['customerPhone'] ?? '',
        'saathiId': data['assignedDriverId'] ?? '',
        'pickupVillage': data['pickupVillage'] ?? '',
        'pickupLat': data['pickupLat'] ?? 0.0,
        'pickupLng': data['pickupLng'] ?? 0.0,
        'destinationVillage': data['destinationVillage'] ?? '',
        'status': data['status'] ?? 'cancelled',
        'fare': data['fare'] ?? 0,
        'cancelReason': data['cancelReason'] ?? '',
        'createdAt': data['createdAt'],
        'migratedFrom': 'bookings',
      }, SetOptions(merge: true));
      
    } else if (type == 'haul') {
      // Move to haul_bookings collection
      final haulRef = FirebaseFirestore.instance
          .collection('haul_bookings')
          .doc(doc.id);
      
      batch.set(haulRef, {
        'bookingId': doc.id,
        'customerId': data['userId'] ?? '',
        'vehicleType': data['vehicleType'] ?? '',
        'pickupVillage': data['pickupVillage'] ?? '',
        'pickupLat': data['pickupLat'] ?? 0.0,
        'pickupLng': data['pickupLng'] ?? 0.0,
        'status': data['status'] ?? 'cancelled',
        'appCommission': 75,
        'createdAt': data['createdAt'],
        'migratedFrom': 'bookings',
      }, SetOptions(merge: true));
    }
  }
  
  await batch.commit();
  debugPrint('✅ Old bookings migrated to rides + haul_bookings');
}
```

```dart
class RideService {
  static final _db = FirebaseFirestore.instance;
  static final _rides = _db.collection('rides');
  
  // Create new ride request
  static Future<String> createRide({
    required String customerId,
    required String customerName,
    required String customerPhone,
    required String pickupVillage,
    required double pickupLat,
    required double pickupLng,
    required String destinationVillage,
    required double destinationLat,
    required double destinationLng,
    required double fare,
    required double distance,
  }) async {
    final rideRef = _rides.doc();
    final otp = (1000 + Random().nextInt(9000)).toString();
    
    await rideRef.set({
      'rideId': rideRef.id,
      'customerId': customerId,
      'customerName': customerName,
      'customerPhone': customerPhone,
      'saathiId': '',
      'saathiName': '',
      'saathiPhone': '',
      'pickupVillage': pickupVillage,
      'pickupLat': pickupLat,
      'pickupLng': pickupLng,
      'destinationVillage': destinationVillage,
      'destinationLat': destinationLat,
      'destinationLng': destinationLng,
      'status': 'searching',
      'fare': fare,
      'distance': distance,
      'otp': otp,
      'saathiLat': 0.0,
      'saathiLng': 0.0,
      'cancelReason': '',
      'rating': 0,
      'createdAt': FieldValue.serverTimestamp(),
    });
    
    return rideRef.id;
  }
  
  // Saathi accepts ride
  static Future<void> acceptRide({
    required String rideId,
    required String saathiId,
    required String saathiName,
    required String saathiPhone,
  }) async {
    await _rides.doc(rideId).update({
      'saathiId': saathiId,
      'saathiName': saathiName,
      'saathiPhone': saathiPhone,
      'status': 'accepted',
      'acceptedAt': FieldValue.serverTimestamp(),
    });
  }
  
  // Update Saathi live location during ride
  static Future<void> updateSaathiLocation({
    required String rideId,
    required double lat,
    required double lng,
  }) async {
    await _rides.doc(rideId).update({
      'saathiLat': lat,
      'saathiLng': lng,
    });
  }
  
  // Start ride (after OTP verified)
  static Future<void> startRide(String rideId) async {
    await _rides.doc(rideId).update({
      'status': 'started',
      'startedAt': FieldValue.serverTimestamp(),
    });
  }
  
  // Complete ride
  static Future<void> completeRide(String rideId) async {
    await _rides.doc(rideId).update({
      'status': 'completed',
      'completedAt': FieldValue.serverTimestamp(),
    });
    
    // Increment saathi total rides
    final ride = await _rides.doc(rideId).get();
    final saathiId = ride.data()!['saathiId'];
    if (saathiId != null && saathiId != '') {
      await FirebaseFirestore.instance
          .collection('saathis')
          .doc(saathiId)
          .update({'totalRides': FieldValue.increment(1)});
    }
  }
  
  // Listen to ride status (real-time)
  static Stream<DocumentSnapshot> listenToRide(String rideId) {
    return _rides.doc(rideId).snapshots();
  }
  
  // Get pending rides for a saathi (incoming requests)
  static Stream<QuerySnapshot> listenForIncomingRides(
      String saathiVillage) {
    return _rides
        .where('status', isEqualTo: 'searching')
        .orderBy('createdAt', descending: true)
        .limit(10)
        .snapshots();
  }
}
```

```dart
class SaathiService {
  static final _saathis = 
      FirebaseFirestore.instance.collection('saathis');
  
  // Go online — start location updates
  static Future<void> goOnline(String uid) async {
    final pos = await Geolocator.getCurrentPosition(
      locationSettings: const LocationSettings(
        accuracy: LocationAccuracy.high,
      ),
    );
    
    final geoPoint = GeoFirePoint(pos.latitude, pos.longitude);
    
    await _saathis.doc(uid).update({
      'isAvailable': true,
      'isOnline': true,
      'position': geoPoint.data,
      'lastSeen': FieldValue.serverTimestamp(),
    });
  }
  
  // Go offline
  static Future<void> goOffline(String uid) async {
    await _saathis.doc(uid).update({
      'isAvailable': false,
      'isOnline': false,
      'lastSeen': FieldValue.serverTimestamp(),
    });
  }
  
  // Update live location (called every 10 sec when online)
  static Future<void> updateLocation(
      String uid, double lat, double lng) async {
    final geoPoint = GeoFirePoint(lat, lng);
    await _saathis.doc(uid).update({
      'position': geoPoint.data,
      'lastSeen': FieldValue.serverTimestamp(),
    });
  }
  
  // Get saathi profile
  static Future<DocumentSnapshot> getSaathi(String uid) {
    return _saathis.doc(uid).get();
  }
}
```

```dart
class VillageService {
  
  static List<VillageModel>? _cachedVillages;
  
  // Hardcoded fallback — correct coordinates
  static final List<VillageModel> _fallbackVillages = [
    VillageModel('Anaval',     'આણવલ',    20.8394, 73.2637),
    VillageModel('Kos',        'કૉસ',     20.8480, 73.2350),
    VillageModel('Tarkani',    'તારકણી',  20.8550, 73.2580),
    VillageModel('Angaldhara', 'અંગળધરા', 20.8180, 73.2280),
    VillageModel('Dholikuva',  'ઢોળીકૂવા',20.8650, 73.2800),
    VillageModel('Lakhavadi',  'લખાવડી',  20.8050, 73.2150),
    VillageModel('Unai',       'ઉનાઈ',    20.8550, 73.2100),
    VillageModel('Doldha',     'ડોળધા',   20.7950, 73.2600),
    VillageModel('Kamboya',    'કાંબોયા', 20.8750, 73.2200),
  ];
  
  static Future<List<VillageModel>> getVillages() async {
    if (_cachedVillages != null) return _cachedVillages!;
    
    try {
      final snapshot = await FirebaseFirestore.instance
          .collection('villages')
          .where('isActive', isEqualTo: true)
          .get();
      
      if (snapshot.docs.isEmpty) {
        _cachedVillages = _fallbackVillages;
        return _fallbackVillages;
      }
      
      _cachedVillages = snapshot.docs.map((doc) {
        final d = doc.data();
        return VillageModel(
          d['name'] ?? '',
          d['nameGu'] ?? '',
          (d['lat'] ?? 0).toDouble(),
          (d['lng'] ?? 0).toDouble(),
        );
      }).toList();
      
      return _cachedVillages!;
      
    } catch (e) {
      // Fallback if Firestore fails
      _cachedVillages = _fallbackVillages;
      return _fallbackVillages;
    }
  }
  
  // Find nearest village to given coordinates
  static VillageModel findNearestVillage(
      double lat, double lng, List<VillageModel> villages) {
    VillageModel nearest = villages.first;
    double minDistance = double.infinity;
    
    for (final village in villages) {
      final distance = Geolocator.distanceBetween(
        lat, lng, village.lat, village.lng,
      );
      if (distance < minDistance) {
        minDistance = distance;
        nearest = village;
      }
    }
    return nearest;
  }
}
```
